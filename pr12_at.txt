import socket

print("1. Enter Host Name")
print("2. Enter IP Address")
choice = int(input("Choice= "))

if choice == 1:
    host = input("\nEnter host name: ")
    try:
        address = socket.gethostbyname(host)
        print("IP Address:", address)
        print("Host Name:", host)
        print("Host name and IP address:", f"{host}/{address}")
    except socket.gaierror:
        print("Could not find host:", host)

elif choice == 2:
    ip = input("\nEnter IP address: ")
    try:
        host_name = socket.gethostbyaddr(ip)[0]
        print("Host Name:", host_name)
        print("IP Address:", ip)
        print("Host name and IP address:", f"{host_name}/{ip}")
    except socket.herror:
        print("Could not find IP address:", ip)

else:
    print("Invalid choice. Please enter 1 or 2.")
